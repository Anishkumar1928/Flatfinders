#user table
create table user(
    Name varchar(20) not null,
    mobile number(13) primary key,
    Email varchar(20) not null,
    Password varchar(20) not null,
    gender varcahr(1) not null check(gender in('M','F','T'))
    Role varchar(7) not null check(Role in('Owner','Renter'))
)

# Room table
create table Room(
Room_id number foreign key(room_id)               reference user(mobile),
rent decimal not null,
dimensions decimal not null,
room_type varchar,
occupancy varchar,
Parking_availability varchar,
)
#Photo table 
create table photo (
owner_id number foreign key(owner_id) reference room(room_id),
photo varchar not null # link of the photo 
)
